        function generateLinks() {
            const input = document.getElementById('inputLinks').value;
            const lines = input.split('\n').filter(line => line.trim() !== '');
            const resultArea = document.getElementById('resultArea');
            
            // Ở đây, thực tế bạn cần gọi API Shopee để convert. 
            // Dưới đây là mô phỏng thêm mã tracking ID của bạn vào link.
            const trackingId = "YOUR_AFFILIATE_ID"; 
            
            const converted = lines.map(link => {
                // Logic giả lập: Thêm tham số affiliate
                return `${link.trim()}?sp_atk=${Math.random().toString(36).substring(7)}&utm_source=aff_tool`;
            });

            resultArea.innerText = converted.join('\n');
        }

        function copyAll() {
            const text = document.getElementById('resultArea').innerText;
            navigator.clipboard.writeText(text);
            alert("Đã sao chép tất cả link!");
        }
